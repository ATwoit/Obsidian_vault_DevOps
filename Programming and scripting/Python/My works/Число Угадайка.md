```python
import random

# объявление функции
def is_valid(num):
    if 1 <= num <= 100:
        if num == ran_num:
            return 'Вы угадали, поздравляем!'
        elif num < ran_num:
            return 'Ваше число меньше загаданного, попробуйте еще разок'
        else:
            return 'Ваше число больше загаданного, попробуйте еще разок'
    else:
        return 'А может быть все-таки введем целое число от 1 до 100?'

# генерируем любое число от 1 до 100
ran_num = random.randint(1, 100)
  

# начинаем игру
print('Добро пожаловать в числовую угадайку!')

while True:

    # считываем данные
    number = int(input('Введите любое число от 1 до 100: '))
    
    # вызываем функцию
    result = is_valid(number)
    print(result)

    # если число угадано, игра завершается
    if result == 'Вы угадали, поздравляем!':
        break
```
